cmake_minimum_required(VERSION 3.10)
project(AtomicCIM LANGUAGE CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)


#location
include_directories(HEADERS_DIR ${CMAKE_SOURCE_DIR}/headers)
include_directories(SRC_DIR ${CMAKE_SOURCE_DIR}/programes)

#list sources::
set(
    SRC_FILES
    ${SRC_DIR}/AtomicCSCompact.cpp
    ${CMAKE_SOURCE_DIR}/main.cpp
)


add_executable(AtomicCIM ${SRC_FILES})
#warning
if(MSVC)
    target_compile_options(AtomicCIM PRIVATE /W4 /WX)
else()
    target_compile_options(AtomicCIM PRIVATE -Wall -Wextra -Wpedantic - Werror)
endif()
